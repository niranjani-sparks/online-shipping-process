package oops;
import java.util.Scanner;
class onlineorder{
	int orderId;
	String customername;
	String productname;
	int quantity;
	double price;
	String orderstatus;
	onlineorder(int a, String b, String c, int d , double e, String f )
	{
		 orderId=a;
		 customername=b;
		 productname=c;
		 quantity=d;
		 price=e;
		 orderstatus=f;
		
	}
	void display() {
		System.out.println("orderId : "+orderId);
		System.out.println("customername : "+customername);
		System.out.println("productname : "+productname);
		System.out.println("quantity : "+quantity);
		System.out.println("price : "+price);
		System.out.println("Totalprice : "+calculationTotalprice(quantity,price));
		System.out.println("orderstatus : "+orderstatus);
	}
	
		static double calculationTotalprice(int quantity,double price) {
			return quantity*price;
		}
}
